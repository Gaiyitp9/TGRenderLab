file(GLOB_RECURSE SRCS *.cpp *.c)
file(GLOB_RECURSE INCS *.hpp *.h)
source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${SRCS})
source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${INCS})

add_library(RenderCore STATIC ${SRCS} ${INCS})
target_include_directories(RenderCore PUBLIC include)
target_link_libraries(RenderCore PRIVATE dxgi d3d11)
if (CMAKE_BUILD_TYPE MATCHES Debug)
	target_link_libraries(RenderCore PRIVATE dxguid dbghelp)
endif()
